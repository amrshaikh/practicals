#p3a

import RPi.GPIO as GPIO
import time
GPIO.setwarning(False)
def blink(pin):
	GPIO.OUTPUT(pin, True)
	time.sleep(0.1)
	GPIO.OUTPUT(pin, False)
	time.sleep(0.1)
GPIO.setmode(GPIO.BOARD)
GPIO.setup(16, GPIO.OUT)
GPIO.setup(18, GPIO.OUT)
for i in range(0, 10):
	blink(16)
	blink(18)
GPIO.cleanup()
